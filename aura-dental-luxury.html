// Client-side sanitization. Demo only; production apps still validate server-side.
function sanitizeInput(raw) {
  if (typeof raw !== 'string') return '';
  return raw
    .replace(/<\s*(script|style)[^>]*>[\s\S]*?<\s*\/\s*\1\s*>/gi, '')
    .replace(/<\/?[^>]+(>|$)/g, '')
    .replace(/javascript\s*:/gi, '')
    .replace(/on\w+\s*=/gi, '')
    .replace(/\s+/g, ' ')
    .trim();
}
